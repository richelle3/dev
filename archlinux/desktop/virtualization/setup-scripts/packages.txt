# Графическая оболочка управления виртуальными машинами
virt-manager

# Низкоуровневый гипервизор / эмулятор (используется libvirt)
qemu

# Служба управления виртуальными машинами
libvirt

# DHCP и DNS сервер, используется для NAT-сетей в libvirt
dnsmasq

# UEFI прошивка для виртуальных машин (для Windows 10/11, Secure Boot и т.д.)
edk2-ovmf

# Утилиты для настройки сетевых мостов (для bridge-сетей)
bridge-utils

# Дополнительные утилиты и инструменты
# Полный набор QEMU с поддержкой всех архитектур, форматов дисков и GUI/headless режимов
qemu-full

# Инструмент для подключения к виртуальной машине (VNC/SPICE), используется virt-manager
virt-viewer

# Набор утилит для работы с виртуальными дисками (просмотр, монтирование, модификация образов)
libguestfs

# Таблицы фильтрации ethernet-кадров, используется libvirt для NAT и bridge-сетей (устаревающий, но ещё встречается)
ebtables

# Virtual Distributed Ethernet — поддержка виртуальных сетей и свитчей (может быть нужен для продвинутой сетевой конфигурации)
vde2

# Версия netcat от OpenBSD, используется QEMU/libvirt для проброса портов и внутренней сетевой связности
openbsd-netcat
